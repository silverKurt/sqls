MEMBER [Measures].[Descricao] AS
    [Measures].[Grupos não comprados], format_string = iif(([Measures].[Descricao] <> 0.0), "Não Comprado", "Comprado")

{[Measures].[Descricao]} ON COLUMNS,
Except([grupoproduto].[Todos].Children, {[grupoproduto].[NAO INFORMADO], [grupoproduto].[SUPER RODO]}) ON ROWS
